create table Orders(
  id int NOT NULL AUTO_INCREMENT,
  userID int,
  primary key (id)
);

create table OrderItems(
  id int NOT NULL AUTO_INCREMENT,
  orderID int,
  listingID int,
  primary key (id)
);

create table Listings (
  id int NOT NULL AUTO_INCREMENT,
  type varchar(50) NOT NULL,
  itemID int NOT NULL,
  primary key (id)
);

create table Carts(
  id int NOT NULL AUTO_INCREMENT,
  listingID int,
  userID int,
  primary key (id)
);
