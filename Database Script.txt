create database if not exists oldegg;
 
use oldegg;
 
drop table if exists Motherboards;
drop table if exists GPUs;
drop table if exists CPUs;
drop table if exists RAMs;
drop table if exists Cases;
drop table if exists Storages;
drop table if exists Coolers;
drop table if exists Users;


create table Motherboards (
  id     int NOT NULL AUTO_INCREMENT,
  brand  varchar(50),
  model varchar(50),
  socket varchar(50),
  price  float,
  qty    int,
  formfactor varchar(50),
  link varchar(150),
  primary key (id)
);

create table GPUs(
  id int NOT NULL AUTO_INCREMENT,
  maker varchar(50),
  brand varchar(50),
  model varchar(50),
  price float,
  qty int,
  link varchar(150),
  primary key (id)
);

create table CPUs(
  id int NOT NULL AUTO_INCREMENT,
  brand varchar(50),
  model varchar(50),
  price float,
  qty int,
  link varchar(150),
  primary key (id)
);

create table Users (
    id int NOT NULL AUTO_INCREMENT,
    email varchar(50),
    name varchar(50),
    password varchar(50),
    mobile_number int,
    primary key (id)
);
